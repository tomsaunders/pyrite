{
	0x0	CHAR[1]	Team

{
	0x000	SHORT	Reserved (1)
	0x002	STR(16)	Name
	0x01A	BOOL[10] Allegiances
	0x024	CHAR[64][6] EndOfMissionMessages

struct Briefing
{
	0x000	SHORT	RunningTime
	0x002	SHORT	Unknown1
	0x004	SHORT	StartEvents
	0x006	INT	EventsLength
	0x00A	Event[] Events
	0x334	Tag[32] Tags
		String[32]
}

struct Event
{
	0x0	SHORT	Time
	0x2	SHORT	Type (enum EventType)
	0x4	SHORT[]	Variables
}

struct Tag
{
	0x0	SHORT	Length
	0x2	CHAR[Length] Tag
}

struct String
{
	0x0	SHORT	Length
	0x2	CHAR[Length] String
}
